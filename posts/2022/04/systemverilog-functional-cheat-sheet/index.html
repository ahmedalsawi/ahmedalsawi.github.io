class foo;

user-defined bins Link to heading 

logic [3:0] color;
covergroup cg @(posedge clk);
cp: coverpoint color {
    bins a = {1,2,3};
    bins b[] = {1,2,3};
    bins c[] = {[1:3]};
}
endgroup

default bins Link to heading 

covergroup cg @(posedge clk);
cp: coverpoint color{
    bins a = {1,2,3};
    bins others[] = default;
}
endgroup

cross Link to heading 

logic [3:0] color;
logic [3:0] color1;

covergroup cg @(posedge clk);
cp: coverpoint color{
    bins a[] = {1,2,3}; // 3 bins
}
cp1: cp, color1 ; // 3 * 16 bins
endgroup

cross bins intersect Link to heading 

logic [3:0] color;
logic [3:0] color1;

covergroup cg @(posedge clk);
cp: coverpoint color{
    bins a[] = {1,2,3};
}
cp1: cp, color1
 {
     bins zeros: binsof(cp) intersect {1}; // <1,0> <1,1> ...
 }
endgroup
